<------------------------ SENSOR CONNECTION TROUBLESHOOTING ------------------------>
# KEY IDEA: the sensor's IP is DHCP, so it CHANGES whenever the network changes.
#   Symptom in the driver log: "Network is unreachable (101, ENETUNREACH)" then "Ouch!"
#   Cause: stream: line in config.yaml points at the sensor's OLD ip.
#   NOTE: 192.168.1.34 / .35 are just THIS LAPTOP (ethernet / wifi), NOT the sensor.
#   NOTE: the wifi you use for the internet is NOT the sensor network unless the
#         sensor is on the same wifi. You can't reach the sensor from a different subnet.

# STEP 1 - which subnet am I on? (note the first 3 octets, e.g. 10.147.182)
ip -4 addr show wlo1 | grep inet

# STEP 2 - find the sensor: scan that subnet for the FP data port 21000
#   (replace 10.147.182 with your subnet from step 1)
for h in $(seq 1 254); do (timeout 1 bash -c "echo > /dev/tcp/10.147.182.$h/21000" 2>/dev/null && echo "SENSOR at 10.147.182.$h") & done; wait

# STEP 3 - confirm it streams FP data (Ctrl-C after you see $FP,... lines)
nc <sensor-ip> 21000

# STEP 4 - set that ip in the stream: line of BOTH files (source + installed copy):
#   fixposition_driver_ros2/launch/config.yaml
#   install/fixposition_driver_ros2/share/fixposition_driver_ros2/launch/config.yaml
#   e.g.  stream: tcpcli://<sensor-ip>:21000

# STEP 5 - launch
ros2 launch fixposition_driver_ros2 node.launch
